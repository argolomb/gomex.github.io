Iniciando ambientes customizados

Especificando ambientes


web:
    image: tutum/apache-php
    environment:
    - VIRTUAL_HOST=site.ufba.br
proxy:
    image: jwilder/nginx-proxy
    volumes:
    - "/var/run/docker.sock:/tmp/docker.sock"
    ports:
    - "80:80"
